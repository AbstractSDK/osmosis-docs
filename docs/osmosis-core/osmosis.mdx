import { useState, useEffect } from 'react'

export const AssetData = () => {
  const [data, setAssets] = useState(null)
  useEffect(() => {
    const fetchAssets = () => {
      fetch('https://raw.githubusercontent.com/cosmos/chain-registry/master/osmosis/chain.json')
        .then((result) => result.json())
        .then((data) => { 
        
if (Array.isArray(data)) {
          data.forEach((item) => {
            console.log(item);
          });
         
        }
         setAssets(data)
        }
        )
    }
    fetchAssets()
  }, [])
  if (!data) {
    return <h1>Fetching Assets...</h1>
  }
return (
  <div>
    [ITERARTE OVER DATA]]
  </div>
)
}
